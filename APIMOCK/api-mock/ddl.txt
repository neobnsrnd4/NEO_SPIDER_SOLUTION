CREATE TABLE `FWK_MOCK_REST_API` (
  `MOCK_ID` int NOT NULL AUTO_INCREMENT COMMENT '순번',
  `MOCK_WIREMOCK_ID` VARCHAR(50) NOT NULL COMMENT 'wiremock uuid',
  `MOCK_API_NAME` varchar(255) NOT NULL COMMENT 'API 이름',
  `MOCK_API_URL` varchar(500) NOT NULL COMMENT 'API 주소',
  `MOCK_API_REQUEST_METHOD` varchar(50) NOT NULL COMMENT 'request method',
  `MOCK_LAST_CHECKED_TIME` datetime DEFAULT NULL COMMENT '마지막 서버 동작 확인 시간',
  `MOCK_LAST_CHECKED_STATUS` int DEFAULT NULL COMMENT '마지막 서버 동작 상태(0:정상,1:장애,2:다운,3:지연)',
  `MOCK_RESPONSE_STATUS` tinyint(1) NOT NULL DEFAULT '1' COMMENT '응답 상태(0:OFF/대응답,1:ON/실서버)',
  PRIMARY KEY (`MOCK_ID`)
);
